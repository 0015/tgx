37// ********************************************************************************
38// Color types.
39//
40// The following color types are availaible for use with the tgx library:
41//
42// - RGB565: 16bits, R:5 G:6 B:5 colors
43// no alpha channel wrapper around a uint16_t integer, aligned as uint16_t
44//
45// - RGB24: 24bits, R:8 G:8 B:8 colors.
46// no alpha channel, not aligned in memory
47//
48// - RGB32: 32bits, R:8 G:8 B:8 A:8 colors.
49// alpha channel, wrapper around a uint32_t integer, aligned as uint32_t
50//
51// - RGB64: 64bits, R:16 G:16 B:16 A:16 colors.
52// alpha channel, wrapper around a uint64_t integer, aligned as uint64_t
53//
54// - RGBf: 96bits, R:float G:float, B:float
55// no alpha channel, aligned as float.
56//
57// - HSV: 96bits, H:float, S:float, V:float
58// hue / saturation / value color space: very slow
59//
60//
61// REMARKS:
62//
63// 1. RGB565, RGB32 and RGB64 are wrappers around basic integer types they can be
64// used as drop in remplacment of uint16_t, uint32_t and uint64_t without any
65// speed penalty.
66//
67// For example:
68// tgx::RGB565 col(12,63,30)
69// uint16_t v = (uint16_t)col; // <- conversion to uint16_t
70// tgx::RGB565 col2 = tgx::RGB565(v) // <- conversion back to RGB565
71//
72// 2. RGB32 and RGB64 have an alpha channel. The color are always assumed to have
73// pre-multiplied alpha.
74//
75// 3. Fast conversion is implemented between color types (and also integer types)
76// except when converting from/to HSV which is slow.
77//
78// ********************************************************************************

499 inline void blend256(const RGB565 & fg_col, uint32_t alpha)
500 {
501 const uint32_t a = (alpha >> 3); // map to 0 - 32.
502 const uint32_t bg = (val | (val << 16)) & 0b00000111111000001111100000011111;
503 const uint32_t fg = (fg_col.val | (fg_col.val << 16)) & 0b00000111111000001111100000011111;
504 const uint32_t result = ((((fg - bg) * a) >> 5) + bg) & 0b00000111111000001111100000011111;
505 val = (uint16_t)((result >> 16) | result); // contract result
506 }

574 inline RGB565 interpolateColorsTriangle(const RGB565 & col1, int32_t C1, const RGB565 & col2, int32_t C2, const RGB565 & col3, const int32_t totC)
575 {
576 C1 <<= 5;
577 C1 /= totC;
578 C2 <<= 5;
579 C2 /= totC;
580 const uint32_t bg1 = (col1.val | (col1.val << 16)) & 0b00000111111000001111100000011111;
581 const uint32_t bg2 = (col2.val | (col2.val << 16)) & 0b00000111111000001111100000011111;
582 const uint32_t bg3 = (col3.val | (col3.val << 16)) & 0b00000111111000001111100000011111;
583 const uint32_t result = ((bg1*C1 + bg2 * C2 + bg3*(32 - C1 - C2)) >> 5) & 0b00000111111000001111100000011111;
584 return RGB565((uint16_t)((result >> 16) | result)); // contract result
585 }

604 inline RGB565 interpolateColorsBilinear(const RGB565 & C00, const RGB565 & C10, const RGB565 & C01, const RGB565 & C11, const float ax, const float ay)
605 {
606 /* flotating point version, slower...
607 const float rax = 1.0f - ax;
608 const float ray = 1.0f - ay;
609 const int R = (int)(rax*(ray*C00.R + ay*C01.R) + ax*(ray*C10.R + ay*C11.R));
610 const int G = (int)(rax*(ray*C00.G + ay*C01.G) + ax*(ray*C10.G + ay*C11.G));
611 const int B = (int)(rax*(ray*C00.B + ay*C01.B) + ax*(ray*C10.B + ay*C11.B));
612 return RGB565(R,G,B);
613 */
614 const int iax = (int)(ax * 256);
615 const int iay = (int)(ay * 256);
616 const int rax = 256 - iax;
617 const int ray = 256 - iay;
618 const int R = rax*(ray*C00.R + iay*C01.R) + iax*(ray*C10.R + iay*C11.R);
619 const int G = rax*(ray*C00.G + iay*C01.G) + iax*(ray*C10.G + iay*C11.G);
620 const int B = rax*(ray*C00.B + iay*C01.B) + iax*(ray*C10.B + iay*C11.B);
621 return RGB565(R >> 16,G >> 16,B >> 16);
622 }
