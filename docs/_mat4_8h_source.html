183 void setOrtho(T left, T right, T bottom, T top, T zNear, T zFar)
184 {
185 memset(M, 0, 16 * sizeof(T));
186 M[0] = ((T)2) / (right - left);

188 M[10] = -((T)2) / (zFar - zNear);
189 M[12] = (right + left) / (left - right);
190 M[13] = (top + bottom) / (bottom - top);

228 void setPerspective(T fovy, T aspect, T zNear, T zFar)
229 {
230 static const T deg2rad = (T)(M_PI / 180);
231 const T aux = tan((fovy / ((T)2)) * deg2rad);
232 const T top = zNear * aux;
233 const T bottom = -top;
234 const T right = zNear * aspect * aux;
235 const T left = -right;
236 setFrustum(left, right, bottom, top, zNear, zFar);

248 void setRotate(T angle, T x, T y, T z)
249 {
250 static const T deg2rad = (T)(M_PI / 180);
251 const T norm = tgx::precise_sqrt(x*x + y*y + z*z);
252 if (norm == 0) { setIdentity(); return; }
253 const T nx = x / norm;
254 const T ny = y / norm;
255 const T nz = z / norm;
256 const T c = cos(deg2rad * angle);
257 const T oneminusc = ((T)1) - c;
258 const T s = sin(deg2rad * angle);
259
260 memset(M, 0, 16 * sizeof(T));
261 M[0] = nx * nx * oneminusc + c;
262 M[1] = ny * nx * oneminusc + nz * s;
263 M[2] = nx * nz * oneminusc - ny * s;
264 M[4] = nx * ny * oneminusc - nz * s;
265 M[5] = ny * ny * oneminusc + c;
266 M[6] = ny * nz * oneminusc + nx * s;
267 M[8] = nx * nz * oneminusc + ny * s;
268 M[9] = ny * nz * oneminusc - nx * s;
269 M[10] = nz * nz * oneminusc + c;
270 M[15] = (T)1;
271 }

464 void setLookAt(T eyeX, T eyeY, T eyeZ, T centerX, T centerY, T centerZ, T upX, T upY, T upZ)
465 {
466 const Vec4<T> f = normalize(Vec4<T>{ centerX - eyeX, centerY - eyeY, centerZ - eyeZ, (T)0 });
467 Vec4<T> up = normalize(Vec4<T>{ upX, upY, upZ, (T)0 });
468 up = normalize(up - (dotProduct(up, f) * f));
469 const Vec4<T> s = crossProduct(f, up);
470 const Vec4<T> u = crossProduct(s, f);
471 M[0] = s.x; M[4] = s.y; M[8] = s.z; M[12] = -s.x * eyeX - s.y * eyeY - s.z * eyeZ;
472 M[1] = u.x; M[5] = u.y; M[9] = u.z; M[13] = -u.x * eyeX - u.y * eyeY - u.z * eyeZ;
473 M[2] = -f.x; M[6] = -f.y; M[10] = -f.z; M[14] = f.x * eyeX + f.y * eyeY + f.z * eyeZ;
474 M[3] = 0; M[7] = 0; M[11] = 0; M[15] = (T)1;
475 }

604 template<typename T> inline Mat4<T> operator*(const Mat4<T> & A, const Mat4<T> & B)
605 {
606 Mat4<T> R;
607 for (int i = 0; i < 4; i++)
608 {
609 for (int j = 0; j < 4; j++)
610 {
611 R.M[i + j*4] = (T)0;
612 for (int k = 0; k < 4; k++) { R.M[i + j * 4] += A.M[i + k * 4] * B.M[k + j * 4]; }
613 }
614 }
615 return R;
616 }
