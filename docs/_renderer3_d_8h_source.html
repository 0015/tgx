1784 template<bool CHECKRANGE, bool USE_BLENDING> TGX_INLINE inline void drawPixelZbuf(int x, int y, color_t color, float opacity, float z)
1785 {
1786 if (CHECKRANGE && ((x < 0) || (x >= _uni.im->lx()) || (y < 0) || (y >= _uni.im->ly()))) return;
1787 ZBUFFER_t& W = _uni.zbuf[x + _uni.im->lx() * y];
1788 const ZBUFFER_t aa = (std::is_same<ZBUFFER_t, uint16_t>::value) ? ((ZBUFFER_t)(z * _uni.wa + _uni.wb)) : ((ZBUFFER_t)z);
1789 if (W < aa)
1790 {
1791 W = aa;
1792 if (USE_BLENDING) _uni.im->template drawPixel<false>(x, y, color, opacity); else _uni.im->template drawPixel<false>(x, y, color);
1793 }
1794 }

1830 void _clip(int & fl, const fVec4 & P, float bx, float Bx, float by, float By)
1831 {
1832 if (P.x >= bx) { fl &= (~(1)); }
1833 if (P.x <= Bx) { fl &= (~(2)); }
1834 if (P.y >= by) { fl &= (~(4)); }
1835 if (P.y <= By) { fl &= (~(8)); }
1836 if ((P.z >= -1.0f)&&(P.w > 0)) { fl &= (~(16)); }
1837 if (P.z <= +1.0f) { fl &= (~(32)); }
1838 }
