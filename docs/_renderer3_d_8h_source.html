48 /*****************************************************************************************
49 ******************************************************************************************
50 *
51 * Drawing methods
52 *
53 ******************************************************************************************
54 *
55 * Methods to draw primitives (triangles/quads/meshes) onto the viewport.
56 *
57 * (1) The rasterizer uses the shaders set with setShaders(). If the requested shaders cannot
58 * be used because they are disabled in the template parameter DISABLED_SHADERS or because
59 * the correct parameters are not supplied (e.g. no normals for Gouraud shading or no image
60 * for texturing) then the drawing operation silently fails and return without drawing anything
61 * (in the best case, it may fall back using another shader).
62 *
63 * -> *** NORMAL VECTORS MUST ALWAYS BE NORMALIZED (UNIT LENGHT) ***
64 *
65 * -> *** TEXTURE DIMENSIONS MUST BE POWERS OF 2 WHEN USING 'FAST' WRAP MODE FOR TEXTURING ****
66 *
67 *
68 * (2) Depth testing is automatically performed when a z-buffer is available.
69 *
70 * -> *** DO NOT FORGET TO CLEAR THE ZBUFFER WITH clearZbuffer() BEFORE DRAWING A NEW SCENE ***
71 *
72 *
73 * (3) Back face culling is automatically performed if enabled (via the setCulling() method)
74 *
75 * -> *** TRIANGLES/QUADS MUST BE GIVEN TO THE DRAWING METHOD IN THEIR CORRECT WINDING ORDER ***
76 *
77 *
78 * (4) It is much more efficient to use methods that draws several primitives at once than issuing
79 * multiple drawing commands for single triangles/quads. Use the following methods by order of
80 * preferences:
81 *
82 * 1 - drawMesh() is the 'most optimized' method and the one that will usually give the faster
83 * rendering so it should be used whenever possible (for static geometry).
84 *
85 * 3 - and drawQuads() Not as fast a drawing a mesh.
86 *
87 * 3 - drawTriangles() Even slower than the methods above.
88 *
89 * 4 - drawTriangle() / drawQuad() Should only be used for drawing single triangles/quads
90 * slowest method because each draw call has an additional
91 * overhead.
92 *
93 ******************************************************************************************
94 ******************************************************************************************/

1817 template<bool CHECKRANGE, bool USE_BLENDING> TGX_INLINE inline void drawPixelZbuf(int x, int y, color_t color, float opacity, float z)
1818 {
1819 if (CHECKRANGE && ((x < 0) || (x >= _uni.im->lx()) || (y < 0) || (y >= _uni.im->ly()))) return;
1820 ZBUFFER_t& W = _uni.zbuf[x + _uni.im->lx() * y];
1821 const ZBUFFER_t aa = (std::is_same<ZBUFFER_t, uint16_t>::value) ? ((ZBUFFER_t)(z * _uni.wa + _uni.wb)) : ((ZBUFFER_t)z);
1822 if (W < aa)
1823 {
1824 W = aa;
1825 if (USE_BLENDING) _uni.im->template drawPixel<false>(x, y, color, opacity); else _uni.im->template drawPixel<false>(x, y, color);
1826 }
1827 }
