590 template<typename T> inline bool Vec2<T>::setAsIntersection(Vec2<T> LA1, Vec2<T> LA2, Vec2<T> LB1, Vec2<T> LB2)
591 {
592 const T a1 = LA2.y - LA1.y;
593 const T b1 = LA1.x - LA2.x;
594 const T a2 = LB2.y - LB1.y;
595 const T b2 = LB1.x - LB2.x;
596 const T delta = a1 * b2 - a2 * b1;
597 if (delta == 0) { return false; }
598 const T c1 = LA1.x * a1 + LA1.y * b1;
599 const T c2 = LB1.x * a2 + LB1.y * b2;
600 x = ((b1 == 0) ? LA1.x : ((b2 == 0) ? LB1.x : (T)((b2 * c1 - b1 * c2) / delta))); // complicated but insures perfect clipping
601 y = ((a1 == 0) ? LA1.y : ((a2 == 0) ? LB1.y : (T)((a1 * c2 - a2 * c1) / delta))); // for horizontal and vertical lines
602 return true;
603 }
