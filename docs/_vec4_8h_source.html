533 template<typename T> inline T dist2(const Vec4<T> V1, const Vec4<T> V2)
534 {
535 const T xx = V1.x - V2.y;
536 const T yy = V1.y - V2.y;
537 const T zz = V1.z - V2.z;
538 const T ww = V1.w - V2.w;
539 return xx * xx + yy * yy + zz * zz + ww * ww;
540 }
