507 Box2<T> getSplit(int part) const
508 {
509 const T midX = (minX + maxX)/2;
510 const T midY = (minY + maxY)/2;
511 switch (part)
512 {
513 case TGX_BOX2_SPLIT_UP: { return Box2<T>(minX, maxX, midY, maxY); }
514 case TGX_BOX2_SPLIT_DOWN: { return Box2<T>(minX, maxX, minY, midY); }
515 case TGX_BOX2_SPLIT_LEFT: { return Box2<T>(minX, midX, minY, maxY); }
516 case TGX_BOX2_SPLIT_RIGHT: { return Box2<T>(midX, maxX, minY, maxY); }
517 case TGX_BOX2_SPLIT_UP_LEFT: { return Box2<T>(minX, midX, midY, maxY); }
518 case TGX_BOX2_SPLIT_UP_RIGHT: { return Box2<T>(midX, maxX, midY, maxY); }
519 case TGX_BOX2_SPLIT_DOWN_LEFT: { return Box2<T>(minX, midX, minY, midY); }
520 case TGX_BOX2_SPLIT_DOWN_RIGHT: { return Box2<T>(midX, maxX, minY, midY); }
521 }
522 return *this;
523 }

537 Vec2<T> getAnchor(int anchor_pos) const
538 {
539 switch (anchor_pos)
540 {
541 case TGX_BOX2_ANCHOR_TOPLEFT: return Vec2<T>(minX, maxY);
542 case TGX_BOX2_ANCHOR_TOPRIGHT: return Vec2<T>(maxX, maxY);
543 case TGX_BOX2_ANCHOR_BOTTOMLEFT: return Vec2<T>(minX, minY);
544 case TGX_BOX2_ANCHOR_BOTTOMRIGHT: return Vec2<T>(maxX, minY);
545 case TGX_BOX2_ANCHOR_CENTER: return Vec2<T>((minX + maxX) / 2, (minY + maxY) / 2);
546 case TGX_BOX2_ANCHOR_CENTERLEFT: return Vec2<T>(minX, (minY + maxY) / 2);
547 case TGX_BOX2_ANCHOR_CENTERRIGHT: return Vec2<T>(maxX, (minY + maxY) / 2);
548 case TGX_BOX2_ANCHOR_CENTERTOP: return Vec2<T>((minX + maxX) / 2, maxY);
549 case TGX_BOX2_ANCHOR_CENTERBOTTOM: return Vec2<T>((minX + maxX) / 2, minY);
550 }
551 return Vec2<T>((minX + maxX) / 2, (minY + maxY) / 2); // unknown pos returns center.
552 }

685 template<typename T> Box2<T> Box2<T>::getEnclosedWithSameRatioAs(const Box2<T> & B) const
686 {
687 Box2<T> C;
688 if (ratio() < B.ratio())
689 {
690 C.minX = 0;
691 C.maxX = maxX - minX;
692 T ll = (T)(lx() / B.ratio());
693 C.minY = (ly() - ll) / 2;
694 C.maxY = C.minY + ll - (std::is_integral<T>::value ? 1 : 0);
695 }
696 else
697 {
698 C.minY = 0;
699 C.maxY = maxY - minY;
700 T ll = (T)(ly() * B.ratio());
701 C.minX = (lx() - ll) / 2;
702 C.maxX = C.minX + ll - (std::is_integral<T>::value ? 1 : 0);
703 }
704 return C;
705 }
