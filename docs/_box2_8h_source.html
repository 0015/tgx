556 Box2<T> getSplit(BOX_SPLIT part) const
557 {
558 const T midX = (minX + maxX)/2;
559 const T midY = (minY + maxY)/2;
560 switch (part)
561 {
562 case SPLIT_TOP: { return Box2<T>(minX, maxX, midY, maxY); }
563 case SPLIT_BOTTOM: { return Box2<T>(minX, maxX, minY, midY); }
564 case SPLIT_LEFT: { return Box2<T>(minX, midX, minY, maxY); }
565 case SPLIT_RIGHT: { return Box2<T>(midX, maxX, minY, maxY); }
566 case SPLIT_TOPLEFT: { return Box2<T>(minX, midX, midY, maxY); }
567 case SPLIT_TOPRIGHT: { return Box2<T>(midX, maxX, midY, maxY); }
568 case SPLIT_BOTTOMLEFT: { return Box2<T>(minX, midX, minY, midY); }
569 case SPLIT_BOTTOMRIGHT: { return Box2<T>(midX, maxX, minY, midY); }
570 }

580 Vec2<T> getAnchor(ANCHOR_LOCATION anchor_pos) const
581 {
582 anchor_pos &= (CENTER | LEFT | RIGHT | TOP | BOTTOM); // remove baseline flag
583 switch (anchor_pos)
584 {
585 case TOPLEFT: return Vec2<T>(minX, maxY);
586 case TOPRIGHT: return Vec2<T>(maxX, maxY);
587 case BOTTOMLEFT: return Vec2<T>(minX, minY);
588 case BOTTOMRIGHT: return Vec2<T>(maxX, minY);
589 case CENTER: return Vec2<T>((minX + maxX) / 2, (minY + maxY) / 2);
590 case CENTERLEFT: return Vec2<T>(minX, (minY + maxY) / 2);
591 case CENTERRIGHT: return Vec2<T>(maxX, (minY + maxY) / 2);
592 case CENTERTOP: return Vec2<T>((minX + maxX) / 2, maxY);
593 case CENTERBOTTOM: return Vec2<T>((minX + maxX) / 2, minY);
594 }
595 return Vec2<T>((minX + maxX) / 2, (minY + maxY) / 2); // unknown pos returns center.
596 }

729 template<typename T> Box2<T> Box2<T>::getEnclosedWithSameRatioAs(const Box2<T> & B) const
730 {
731 Box2<T> C;
732 if (ratio() < B.ratio())
733 {
734 C.minX = 0;
735 C.maxX = maxX - minX;
736 T ll = (T)(lx() / B.ratio());
737 C.minY = (ly() - ll) / 2;
738 C.maxY = C.minY + ll - (std::is_integral<T>::value ? 1 : 0);
739 }
740 else
741 {
742 C.minY = 0;
743 C.maxY = maxY - minY;
744 T ll = (T)(ly() * B.ratio());
745 C.minX = (lx() - ll) / 2;
746 C.maxX = C.minX + ll - (std::is_integral<T>::value ? 1 : 0);
747 }
748 return C;
749 }
