550 Box2<T> getSplit(BoxSplit part) const
551 {
552 const T midX = (minX + maxX)/2;
553 const T midY = (minY + maxY)/2;
554 switch (part)
555 {
556 case SPLIT_TOP: { return Box2<T>(minX, maxX, midY, maxY); }
557 case SPLIT_BOTTOM: { return Box2<T>(minX, maxX, minY, midY); }
558 case SPLIT_LEFT: { return Box2<T>(minX, midX, minY, maxY); }
559 case SPLIT_RIGHT: { return Box2<T>(midX, maxX, minY, maxY); }
560 case SPLIT_TOPLEFT: { return Box2<T>(minX, midX, midY, maxY); }
561 case SPLIT_TOPRIGHT: { return Box2<T>(midX, maxX, midY, maxY); }
562 case SPLIT_BOTTOMLEFT: { return Box2<T>(minX, midX, minY, midY); }
563 case SPLIT_BOTTOMRIGHT: { return Box2<T>(midX, maxX, minY, midY); }
564 }
565 return *this;

574 Vec2<T> getAnchor(Anchor anchor_pos) const
575 {
576 anchor_pos &= (CENTER | LEFT | RIGHT | TOP | BOTTOM); // remove baseline flag
577 switch (anchor_pos)
578 {
579 case TOPLEFT: return Vec2<T>(minX, maxY);
580 case TOPRIGHT: return Vec2<T>(maxX, maxY);
581 case BOTTOMLEFT: return Vec2<T>(minX, minY);
582 case BOTTOMRIGHT: return Vec2<T>(maxX, minY);
583 case CENTER: return Vec2<T>((minX + maxX) / 2, (minY + maxY) / 2);
584 case CENTERLEFT: return Vec2<T>(minX, (minY + maxY) / 2);
585 case CENTERRIGHT: return Vec2<T>(maxX, (minY + maxY) / 2);
586 case CENTERTOP: return Vec2<T>((minX + maxX) / 2, maxY);
587 case CENTERBOTTOM: return Vec2<T>((minX + maxX) / 2, minY);
588 }
589 return Vec2<T>((minX + maxX) / 2, (minY + maxY) / 2); // unknown pos returns center.
590 }

723 template<typename T> Box2<T> Box2<T>::getEnclosedWithSameRatioAs(const Box2<T> & B) const
724 {
725 Box2<T> C;
726 if (ratio() < B.ratio())
727 {
728 C.minX = 0;
729 C.maxX = maxX - minX;
730 T ll = (T)(lx() / B.ratio());
731 C.minY = (ly() - ll) / 2;
732 C.maxY = C.minY + ll - (std::is_integral<T>::value ? 1 : 0);
733 }
734 else
735 {
736 C.minY = 0;
737 C.maxY = maxY - minY;
738 T ll = (T)(ly() * B.ratio());
739 C.minX = (lx() - ll) / 2;
740 C.maxX = C.minX + ll - (std::is_integral<T>::value ? 1 : 0);
741 }
742 return C;
743 }
